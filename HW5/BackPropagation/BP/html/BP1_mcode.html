1function [w_init, err_1, w_final, err_final, times]=BP(x, t, rate, max_times, target)
 2% Initialize v's and w's
 3w = 2*rand(1,2,'double') - 1;
 4w_init = w;
 5v = 2*rand(2,2,'double') - 1;
 6hnode = zeros(1,2);

 8v_delta = zeros(2,2);
 9times = 0;
10err_1 = 0;
11w_final = zeros(1,2);

35        break;
36    end
37    % calculate delta_w

47    v = v + v_delta;
48end
